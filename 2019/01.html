import numpy as np
from matplotlib import pyplot as plt
from matplotlib import animation

import random

def _update(frame):
    plt.cla()
    data = []
    for i in range(10000):
        xi = [ random.randint(1, 6) for i in range(frame) ]
        avg = sum(xi) / float(len(xi))
        data.append(avg)
    plt.hist(data, 125, histtype='bar')
 
fig = plt.figure(figsize=(5, 3))
params = {
    'fig': fig, 'func': _update, 'interval': 500, 'frames': [ i+1 for i in range(10) ], 'repeat': False,
}
anime = animation.FuncAnimation(**params)

import numpy as np
from matplotlib import pyplot as plt
from matplotlib import animation

import pandas as pd
from scipy.special import comb

import random

def sim(n, p, x):
    data = list(np.random.binomial(n, p, 10000))
    return data.count(x) / float(len(data))

def _update(frame):
    n = 100
    p = frame / float(10)
    plt.cla()
    xList = pd.Series([comb(float(n), x) * p ** x * (1-p) ** (float(n) - x) for x in range(0, n+1)])
    plt.subplot(1,2,1)
    plt.bar(xList.index, xList)
    plt.xlim(-10, 110)
    plt.xticks(np.arange(0, n+1, 10))
    xList = pd.Series([sim(n, p, x) for x in range(0, n+1)])
    plt.subplot(1,2,2)
    plt.bar(xList.index, xList)
    plt.xlim(-10, 110)
    plt.xticks(np.arange(0, n+1, 10))

fig = plt.figure(figsize=(10, 3))
params = {
    'fig': fig, 'func': _update, 'interval': 500, 'frames': [ i+1 for i in range(9) ], 'repeat': False,
}
anime = animation.FuncAnimation(**params)

import numpy as np
from matplotlib import pyplot as plt
from matplotlib import animation

import random

def _update(frame):
    plt.cla()
    n = 10000
    v = np.zeros(n)
    for i in range(frame):
        nrm = np.random.normal(0, 1, n)
        v += (nrm ** 2)
    plt.subplot(1,2,1)
    plt.hist(v, 60, histtype='bar')
    plt.subplot(1,2,2)
